# Inicializa un repositorio de Git local
git init -b main

# Añade todos los archivos al repositorio
git add .

# Crea tu primer commit (un guardado)
git commit -m "Primer commit del proyecto TestOpos"

# Conecta tu repositorio local con el de GitHub
git remote add origin https://github.com/tu-usuario/tu-repositorio.git

# Sube los archivos a GitHub
git push -u origin main
